1. Array Indicizzati

<? php
// Creazione array
$frutti = array("Mela", "Pera", "Banana");

// Sintassi alternativa (PHP 5.4+)
$colori = ["Rosso", "Verde", "Blu"];

// Accesso elementi (indice parte da 0)
echo $frutti[0];  // Mela
echo $frutti[1];  // Pera
echo $frutti[2];  // Banana

// Aggiungere elementi
$frutti[] = "Arancia";  // Aggiunge in fondo
$frutti[3] = "Uva";     // Specifica indice

// Modificare elementi
$frutti[0] = "Fragola";

// Contare elementi
$lunghezza = count($frutti);
?>

2. Array Associativi

<?php
// Array con chiavi personalizzate
$persona = array(
    "nome" => "Mario",
    "cognome" => "Rossi",
    "eta" => 25,
    "citta" => "Roma"
);

// Sintassi alternativa
$auto = [
    "marca" => "Fiat",
    "modello" => "500",
    "anno" => 2020
];

// Accesso con chiave
echo $persona["nome"];     // Mario
echo $auto["marca"];       // Fiat

// Aggiungere/modificare
$persona["email"] = "mario@example.com";
$persona["eta"] = 26;
?>

3. Array Multidimensionali

<?php
// Array di array
$studenti = array(
    array("Mario", "Rossi", 25),
    array("Anna", "Verdi", 23),
    array("Luca", "Bianchi", 27)
);

// Accesso
echo $studenti[0][0];  // Mario
echo $studenti[1][1];  // Verdi

// Array associativi multidimensionali
$prodotti = array(
    array(
        "nome" => "Laptop",
        "prezzo" => 899.00,
        "disponibile" => true
    ),
    array(
        "nome" => "Mouse",
        "prezzo" => 25.00,
        "disponibile" => false
    )
);

echo $prodotti[0]["nome"];    // Laptop
echo $prodotti[1]["prezzo"];  // 25.00
?>

4. Ciclo Foreach

<?php
// Foreach per array indicizzati
$frutti = ["Mela", "Pera", "Banana"];

foreach ($frutti as $frutto) {
    echo $frutto . " ";
}
// Output: Mela Pera Banana

// Foreach per array associativi
$persona = ["nome" => "Mario", "eta" => 25];

foreach ($persona as $chiave => $valore) {
    echo "$chiave: $valore<br>";
}
// Output:
// nome: Mario
// eta: 25
?>

5. Funzioni Utili per Array

<?php
$numeri = [3, 1, 4, 1, 5, 9];

// Contare elementi
count($numeri);           // 6

// Verificare se esiste
in_array(4, $numeri);     // true
in_array(10, $numeri);    // false

// Ordinamento

rsort($numeri);           // Ordina decrescente
asort($arr);              // Ordina associativo per valore
ksort($arr);              // Ordina associativo per chiave

// Massimo e minimo

// Estrarre slice
