// === uguale e stesso tipo
// !=  diverso

// >   maggiore
// <=  minore o uguale
// >=  maggiore o uguale

// !   NOT logico

3. Istruzioni di Iterazione

<?php
// CICLO FOR
for ($i = 1; $i <= 5; $i++) {
    echo $i . " ";
}
// Output: 1 2 3 4 5

// CICLO WHILE
$contatore = 1;
while ($contatore <= 5) {
    echo $contatore . " ";
    $contatore++;
}

// CICLO DO-WHILE
$n = 1;
do {
    echo $n . " ";
    $n++;
} while ($n <= 5);
?>
